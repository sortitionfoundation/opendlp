{{ _("Signature") }}

stepper(items=[], aria_label="", id="", mode="tabs", preserve_scroll=false)

{{ _("Item keys") }}

  • key: str ✅ ({{ _("unique per stepper — used for tab/panel ids") }})
  • label: str ✅
  • href: str ✅
  • state: "active" | "inactive" | "done" | "error" ❌ ({{ _("derived from `active` if omitted") }})
  • active: bool ❌ ({{ _("shortcut for state='active'") }})
  • disabled: bool ❌ ({{ _("only meaningful in wizard mode") }})

{{ _("Panel convention (same as tabs macro)") }}

                role="tabpanel"
                aria-labelledby="{id}-tab-{key}"
                tabindex="0">
