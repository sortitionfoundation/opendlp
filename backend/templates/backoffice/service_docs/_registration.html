Helper Returns Usage

{# Example Usage #}

{{ _("Example Usage") }}

<form action="{{ form_action }}" method="POST">
                {{ csrf_form_element }}

                {{ form_errors() }}

                <div class="form-group">
                <label for="email">Email</label>
                <input type="email" name="email" id="email"
                value="{{ value('email') }}"
                class="{% if has_error('email') %}error{% endif %}">
                {{ field_errors('email') }}
                </div>

                <div class="form-group">
                <label>Gender</label>
                <input type="radio" name="gender" value="female" {{ checked('gender', 'female') }}> Female
                <input type="radio" name="gender" value="male" {{ checked('gender', 'male') }}> Male
                </div>

                <button type="submit">Register</button>
                </form>

Note: On initial page load (GET request), all helpers return empty values. They are populated with submitted data only after a POST with validation errors.
